git --version	-	версия гита, проверка установлен ли он вообще
git config --global user.name xxxxxxx	-	установка имени пользователя гита глобально
git config --global user.email xxxxxxx	-	установка почты пользователя гита глобально
git config user.name	-	посмореть имя пользователя гита установленное
git init	-	инициализация гита, создаётся папка гита - локальный репозиторий
git add xxx.xx 	-	добавление файла xxx.xx в локальный репозиторий
git add *.xx 	-	добавление файлов с расширением .xx в локальный репозиторий
git add !*.xx 	-	добавление всех файлов кроме с расширением .xx в локальный репозиторий
git add xxx/**/*.xx 	-	добавление файлов с расширением .xx в папке xxx и всех вложенных в неё
git add xxx/ 	-	добавление папки xxx в локальный репозиторий
git add . 	-	добавление всего проекта, содержимого текущей папки в локальный репозиторий
git status	-	статус репозитория, показывает например есть ли ожидающие коммита 
git rm --cached xxx.xx xxx/	-	удаление файла xxx.xx папки xxx из стадии ожидания коммита
git commit -m "комментарий коммита"	-	коммит ожидающих с комментарием
git log	-	история коммитов
git log --oneline	-	история коммитов более сокращённо
git checkout xxxxx 	-	простомтр проекта на момент состояния коммита с id xxxxx(из предыдущей команды)
git checkout master	-	возврат к основной ветке и последнему коммиту
git revert xxxxx 	-	отмена коммита с id xxxxx, сам коммит из истории не удаляется
git reset xxxxx 	-	удаляются все коммиты до коммита с id xxxxx, !файлы не изменяются!
git reset xxxxx --hard	-	удалаются все коммиты до коммита с id xxxxx, !файлы изменяются!

git branch xxx 	-	создание ветки xxx
git branch -a 	-	показать список веток
git checkout xxx 	-	перейти на ветку xxx
git checkout -b xxx 	-	создать и перейти на ветку xxx

git merge xxx 	-	слияние ветки ххх с текущей веткой
